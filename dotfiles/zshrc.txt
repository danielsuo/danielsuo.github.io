# Cloud providers
function ec2-ssh () {
   instance_user="$(cat ~/.ssh/ec2 | jq -r '.'$1'.user')"
   instance_id="$(cat ~/.ssh/ec2 | jq -r '.'$1'.id')"
   instance_key="$(cat ~/.ssh/ec2 | jq -r '.'$1'.key')"

   ssh $instance_user@$(aws ec2 describe-instances --filter Name=instance-id,Values=$instance_id | jq -r '.Reservations[0].Instances[0].PublicIpAddress') -i ~/.ssh/pem/$instance_key.pem
}

# # complete source activate. Thanks to Paul Kienzle from NIST for the
# # suggestion.
_activate_complete ()
{
    local cur="${COMP_WORDS[COMP_CWORD]}";
    COMPREPLY=($(compgen -W "$(ls -C1 $HOME/miniconda3/envs | grep "^[^\.]" | xargs basename | lam -s \" - -s \")" -- "$cur" ));
}

_activate_complete_zsh ()
{
  compadd _activate_complete
}
